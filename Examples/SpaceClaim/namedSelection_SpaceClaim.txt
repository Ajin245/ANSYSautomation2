# ===== НАСТРОЙКИ СКРИПТА =====
GU_FIXED_TYPE = "lowest"  # Варианты: "lowest" - самая нижняя поверхность, 
                          # "perimeter" - грани по периметру тела "podoporka"
# =============================

'''
Макрос для поиска всех уникальных Parts модели и создание Named Selection
по этим партам. В последствии на эти Named Selection можно применять настройки сетки
и обработки результатов.
ВНИМАНИЕ: скрипт работает только с координатной системой
X - прод. вдоль трубы, Y - трав, Z - верт.
'''

def create_named_selections(doc, base_names):
    if not doc or not base_names:
        return
    
    groups = GetActiveWindow().Groups
    root_part = GetRootPart()
    allBodies = root_part.GetAllBodies()
         
    for base_name in base_names:
        pattern_str = "^{0}(?:\s*\d+)?$".format(re.escape(base_name))
        pattern = re.compile(pattern_str, re.IGNORECASE)
        
        matching_bodies = []
        for body in allBodies:
            body_name = body.GetName()
            if body_name and pattern.search(body_name):
                matching_bodies.append(body)
                
        if not matching_bodies:
            continue
        
        matching_selection = Selection.CreateByObjects(matching_bodies)
        new_group = NamedSelection.Create(matching_selection, Selection.Empty())
        gg = new_group.CreatedNamedSelection.SetName(base_name)
                          
        print("Named Selection '{0}' created with {1} bodies".format(base_name, len(matching_bodies)))

def create_gu_fixed_lowest(doc):
    """Создает Named Selection 'gu_fixed' для самой нижней поверхности"""
    if not doc:
        return None
        
    try:
        root_part = GetRootPart()
        allBodies = root_part.GetAllBodies()
        minZCoord = float('inf')
        minZFace = None
        
        for b in allBodies:
            faces = b.Faces
            for f in faces:
                fSel = FaceSelection.Create(f)
                centroid = MeasureHelper.GetCentroid(fSel)
                if centroid.Z < minZCoord:
                    minZCoord = centroid.Z
                    minZFace = fSel
        
        if minZFace:
            minZFace.SetActive()
            new_group = NamedSelection.Create(minZFace, Selection.Empty())
            new_group.CreatedNamedSelection.SetName("gu_fixed")
            print("Named Selection 'gu_fixed' created successfully (lowest surface)")
            return new_group
        else:
            print("No suitable face found for 'gu_fixed'")
            return None
            
    except Exception as ex:
        print("Exception in create_gu_fixed_lowest():", str(ex))
        return None

def create_gu_fixed_perimeter(doc):
    """Создает Named Selection 'gu_fixed' для граней по периметру тела 'podoporka'"""
    if not doc:
        return None
        
    try:
        root_part = GetRootPart()
        allBodies = root_part.GetAllBodies()
        
        # Ищем тело с "podoporka" в имени
        podoporka_body = None
        for body in allBodies:
            if "podoporka" in body.GetName().lower():
                podoporka_body = body
                break
        
        if not podoporka_body:
            print("No bodies with 'podoporka' in name found for perimeter fixed")
            return None
        
        print("Found podoporka body: " + podoporka_body.GetName())
        
        # Получаем все грани тела
        all_faces = []
        for face in podoporka_body.Faces:
            all_faces.append(face)
        
        print("Total faces in body: {}".format(len(all_faces)))
        
        # Вручную вычисляем bounding box тела через центроиды граней
        min_z = float('inf')
        max_z = float('-inf')
        
        for face in all_faces:
            try:
                fSel = FaceSelection.Create(face)
                centroid = MeasureHelper.GetCentroid(fSel)
                
                min_z = min(min_z, centroid.Z)
                max_z = max(max_z, centroid.Z)
                    
            except Exception as e:
                print("Error getting face centroid: " + str(e))
                continue
        
        height = max_z - min_z
        print("Calculated body height: {:.2f} (min_z={:.2f}, max_z={:.2f})".format(height, min_z, max_z))
        
        # Определяем диапазон для боковых граней (исключаем верхние и нижние 25%)
        exclude_percentage = 0.25
        exclude_height = height * exclude_percentage
        bottom_limit = min_z + exclude_height
        top_limit = max_z - exclude_height
        
        print("Middle height range for perimeter: {:.2f} to {:.2f}".format(bottom_limit, top_limit))
        
        # Собираем грани в среднем диапазоне по Z
        candidate_faces = []
        for face in all_faces:
            try:
                fSel = FaceSelection.Create(face)
                centroid = MeasureHelper.GetCentroid(fSel)
                
                if centroid.Z >= bottom_limit and centroid.Z <= top_limit:
                    candidate_faces.append(face)
            except:
                continue
        
        print("Faces in middle height range: {}".format(len(candidate_faces)))
        
        # Фильтрация по площади для исключения отверстий
        if candidate_faces:
            areas = []
            for face in candidate_faces:
                try:
                    sel = FaceSelection.Create(face)
                    area = MeasureHelper.GetArea(sel)
                    areas.append(area)
                except:
                    areas.append(0.0)
            
            avg_area = sum(areas) / len(areas) if areas else 0
            filtered_faces = []
            
            for i, face in enumerate(candidate_faces):
                # Сохраняем грань, если её площадь больше 40% от средней
                if areas[i] >= avg_area * 0.4:
                    filtered_faces.append(face)
            
            print("After area-based filtering: {} faces".format(len(filtered_faces)))
            
            # Если отфильтровали слишком много, берем топ-4 по площади
            if len(filtered_faces) > 4:
                face_area_pairs = []
                for face in filtered_faces:
                    try:
                        sel = FaceSelection.Create(face)
                        area = MeasureHelper.GetArea(sel)
                        face_area_pairs.append((face, area))
                    except:
                        face_area_pairs.append((face, 0.0))
                
                # Сортируем по убыванию площади и берем 4 крупнейших
                face_area_pairs.sort(key=lambda x: x[1], reverse=True)
                filtered_faces = [face for face, area in face_area_pairs[:4]]
                print("Selected 4 largest faces for 'gu_fixed'")
        else:
            filtered_faces = []
        
        # Финальный отбор
        if filtered_faces:
            # Создаем Named Selection
            face_selection = Selection.CreateByObjects(filtered_faces)
            new_group = NamedSelection.Create(face_selection, Selection.Empty())
            new_group.CreatedNamedSelection.SetName("gu_fixed")
            print("Named Selection 'gu_fixed' created successfully with {} faces".format(len(filtered_faces)))
            return new_group
        else:
            print("Could not identify suitable perimeter faces.")
            return None
            
    except Exception as ex:
        print("Exception in create_gu_fixed_perimeter(): " + str(ex))
        import traceback
        traceback.print_exc()
        return None

def create_gu_fixed_ns(doc):
    """Основная функция создания gu_fixed в зависимости от настроек"""
    print("Creating gu_fixed with type: " + GU_FIXED_TYPE)
    
    if GU_FIXED_TYPE == "perimeter":
        return create_gu_fixed_perimeter(doc)
    else:  # "lowest" по умолчанию
        return create_gu_fixed_lowest(doc)

def create_gu_force_ns(doc):
    """Создает Named Selection 'gu_force' в центре масс труб"""
    if not doc:
        return None
        
    try:
        root_part = GetRootPart()
        bodies = root_part.GetAllBodies(r'^truba\d+')
        
        if not bodies:
            print("No bodies with 'truba' prefix found for 'gu_force'")
            return None

        curvePoints = []
        for b in bodies:
            for edge in b.Edges:
                curvePoints.append(edge.GetChildren[ICurvePoint]())
                       
        sel = Selection.CreateByObjects(bodies)
        massCenter = MeasureHelper.GetCenterOfMass(sel)
        myPoint = Geometry.Point.Create(round(massCenter.X,4), round(massCenter.Y,4), round(massCenter.Z,4))
        Selection.Clear()

        forcePoint = []
        for curvePoint in curvePoints:
            for p in curvePoint:
                if (round(p.Position.X,4) == myPoint.Position.X and 
                    round(p.Position.Y,4) == myPoint.Position.Y and 
                    round(p.Position.Z,4) == myPoint.Position.Z):
                    forcePoint.append(p)
                
        if len(forcePoint) >0:
            print("forcePoint located SUCCESSFULLY")
            se = Selection.CreateByObjects(forcePoint[0])
            new_group = NamedSelection.Create(se, Selection.Empty())
            new_group.CreatedNamedSelection.SetName("gu_force")
            return new_group
        else:
            print("No force point found at center of mass")
            return None
            
    except Exception as ex:
        print("Exception in create_gu_force_ns():", str(ex))
        return None

def create_gu_remote_disp_ns(doc):
    """Создает Named Selection 'gu_remote_disp' для торцов труб (исправленная)"""
    if not doc:
        return None
        
    try:
        root_part = GetRootPart()
        
        # Находим все тела с "truba" в имени
        truba_bodies = []
        all_bodies = root_part.GetAllBodies()
        
        for body in all_bodies:
            body_name = body.GetName().lower()
            if "truba" in body_name:
                truba_bodies.append(body)
        
        if not truba_bodies:
            print("No pipe bodies found for 'gu_remote_disp'")
            return None
        
        print("Found {} pipe bodies".format(len(truba_bodies)))
        
        # Собираем данные о всех гранях всех труб
        all_face_data = []
        
        for body in truba_bodies:
            try:
                faces = body.Faces
                
                for face in faces:
                    try:
                        # Создаем FaceSelection для получения центроида
                        fSel = FaceSelection.Create(face)
                        centroid = MeasureHelper.GetCentroid(fSel)
                        
                        # Получаем площадь через свойство Area грани
                        area = face.Area
                        
                        # Сохраняем данные
                        all_face_data.append({
                            'face': face,
                            'body': body,
                            'centroid': centroid,
                            'area': area
                        })
                        
                    except Exception as e:
                        print("Error processing face: {}".format(str(e)))
                        continue
                        
            except Exception as e:
                print("Error processing body {}: {}".format(body.GetName(), str(e)))
                continue
        
        if not all_face_data:
            print("No face data collected")
            return None
        
        print("Collected data for {} faces".format(len(all_face_data)))
        
        # Определяем глобальные границы по X
        x_coords = [data['centroid'].X for data in all_face_data]
        global_min_x = min(x_coords)
        global_max_x = max(x_coords)
        x_range = global_max_x - global_min_x
        
        print("Global X bounds: min={:.3f}, max={:.3f}, range={:.3f}".format(
            global_min_x, global_max_x, x_range))
        
        # Определяем допуск (1% от диапазона)
        tolerance = x_range * 0.01
        
        # Ищем торцевые грани
        end_faces = []
        
        for body in truba_bodies:
            # Фильтруем грани этого тела
            body_faces = [d for d in all_face_data if d['body'] == body]
            
            if not body_faces:
                continue
            
            # Находим грани на левом и правом торце
            left_candidates = []
            right_candidates = []
            
            for data in body_faces:
                centroid = data['centroid']
                face = data['face']
                area = data['area']
                
                # Проверяем, близка ли грань к левому или правому торцу
                if abs(centroid.X - global_min_x) <= tolerance:
                    left_candidates.append((face, area, centroid.X))
                elif abs(centroid.X - global_max_x) <= tolerance:
                    right_candidates.append((face, area, centroid.X))
            
            # Для каждого торца выбираем грань с максимальной площадью
            if left_candidates:
                left_candidates.sort(key=lambda x: x[1], reverse=True)
                best_left = left_candidates[0][0]
                end_faces.append(best_left)
                print("Body {}: selected left face, area={:.3f}, X={:.3f}".format(
                    body.GetName(), left_candidates[0][1], left_candidates[0][2]))
            
            if right_candidates:
                right_candidates.sort(key=lambda x: x[1], reverse=True)
                best_right = right_candidates[0][0]
                end_faces.append(best_right)
                print("Body {}: selected right face, area={:.3f}, X={:.3f}".format(
                    body.GetName(), right_candidates[0][1], right_candidates[0][2]))
        
        if end_faces:
            # Удаляем дубликаты (на всякий случай)
            unique_faces = []
            seen_ids = set()
            
            for face in end_faces:
                face_id = id(face)
                if face_id not in seen_ids:
                    seen_ids.add(face_id)
                    unique_faces.append(face)
            
            print("Selected {} unique end faces".format(len(unique_faces)))
            
            # Создаем Named Selection
            face_selection = Selection.CreateByObjects(unique_faces)
            new_group = NamedSelection.Create(face_selection, Selection.Empty())
            new_group.CreatedNamedSelection.SetName("gu_remote_disp")
            
            print("Named Selection 'gu_remote_disp' created with {} faces".format(len(unique_faces)))
            
            # Выводим детальную информацию
            print("\nDetailed information about selected faces:")
            for i, face in enumerate(unique_faces):
                try:
                    fSel = FaceSelection.Create(face)
                    centroid = MeasureHelper.GetCentroid(fSel)
                    area = face.Area
                    print("  Face {}: area={:.3f}, X={:.3f}, Y={:.3f}, Z={:.3f}".format(
                        i+1, area, centroid.X, centroid.Y, centroid.Z))
                except:
                    print("  Face {}: info unavailable".format(i+1))
            
            return new_group
        else:
            print("No end faces found for 'gu_remote_disp'")
            return None
            
    except Exception as ex:
        print("Exception in create_gu_remote_disp_ns(): " + str(ex))
        import traceback
        traceback.print_exc()
        return None

def search_unique_body_names(doc):
    unique_names = set()
    if not doc:
        return
    try:
        root_part = GetRootPart()
        allBodies = root_part.GetAllBodies()
        for body in allBodies:
            if body.GetName().Split(' ')[0]:
                unique_names.add(body.GetName().Split(' ')[0])
        return sorted(list(unique_names))
    except Exception as ex:
        print("Exception in search_unique_body_names(doc)", str(ex))
        return
